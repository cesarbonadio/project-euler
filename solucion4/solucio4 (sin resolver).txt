#include <stdio.h>
#include <stdlib.h>
#include <iostream>
using namespace std;


int palindrome(int x) {
	int aux = x, digit, twirled = 0;

	while (aux != 0) {
		digit = aux % 10;
		aux = aux / 10;
		twirled = (twirled * 10) + digit;
	 }

	if (twirled == x) return 1;
	else return 0;
}


int even(int x) {
	if (x % 2 == 0) return 1;
	else return 0;
}


int main() {

	int first = 0, num1 = 100, num2 = 100,  cont;
	long int n, total;

	while ((num2 < 999) || (num1 < 999)) {

	    if (even(first)) num1++; 
		else if (!(even(first))) num2++;   

		n = num1*num2;

		if (palindrome(n)) total = n; 
		
		first++;
	}

	cout << total << endl;

	system("pause");
	return 0;
}
