#include <stdio.h>
#include <stdlib.h>
#include <iostream>
using namespace std;


int primo(int x) {
	int i,cont=0;

	for (i = 1; i <= x; i++)
		if (x%i == 0) cont++;

	if (cont > 2) return 0;
	else return 1;
}


int main() {

	long long int i, result = 1; long long int k = 600851475143;

	for (i = 1; i<=k;i++) if ((primo(i)) && (k%i == 0))  result = i;
		
		
	cout << result << endl;

	system("pause");
	return 0;
}
