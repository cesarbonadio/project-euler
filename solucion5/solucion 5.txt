EN PASCAL:

program inicio;
uses crt;

 var
  check:boolean;
  i,k,cont:integer;

BEGIN

 i:= 1;
 k:= 1;

 while (check=false) do
 begin

 k:=k+1;
 cont:= 0;

 for i:=1 to 20 do
     begin
     if (k mod i = 0) then cont:= cont+1
     else break;
     end;

  if (cont=20) then check:=true;


 end;


 writeln (k);

readkey;

END.








EN C:

#include <stdio.h>
#include <stdlib.h>
#include <iostream>
using namespace std;


int main() {
	int i = 1, k = 1, cont = 0;

	while (i) {
		++k;
		cont = 0;

		for (i = 1; i <= 20; i++) {
			if ((k%i) == 0) ++cont;
			else break;
		}

		if (cont == 20) { cout << k << endl; break; }
	}

	system("pause");
	return 0;
}
